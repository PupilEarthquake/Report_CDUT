\ProvidesFile{Title_set.tex}[2023]

\RequirePackage{titlesec}

\titleformat{name=\chapter}
    [block]
    {\centering\Huge\sffamily\heiti}
    {第\,\thechapter\,章}
    {1em}
    {}
\titlespacing*{\chapter}{0pt}{0pt}{40pt}  % 去除chapter上方的间距
  

\titleformat{name=\section}
    [block]
    {\LARGE\sffamily\CJKfamily{heiti}}
    {\thesection}
    {1em}
    {}
    []

\titleformat{name=\subsection}
    [block]
    {\Large\sffamily\CJKfamily{heiti}}
    {\thesubsection}
    {1em}
    {}
    []

\titleformat{name=\subsubsection}
    [block]
    {\large\sffamily\CJKfamily{heiti}}
    {\thesubsubsection}
    {1em}
    {}
    []
